5   package oshi.driver.windows.wmi;
6   
7   import java.util.Objects;
8   
9   import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiQuery;
10  import com.sun.jna.platform.win32.COM.WbemcliUtil.WmiResult;
11  
12  import oshi.annotation.concurrent.ThreadSafe;
13  import oshi.util.platform.windows.WmiQueryHandler;
14  
15  /**
16   * Utility to query WMI class {@code Win32_BaseBoard}
17   */
18  @ThreadSafe
19  public final class Win32BaseBoard {
20  
21      private static final String WIN32_BASEBOARD = "Win32_BaseBoard";
22  
23      /**
24       * Baseboard description properties.
25       */
26      public enum BaseBoardProperty {
27          MANUFACTURER, MODEL, PRODUCT, VERSION, SERIALNUMBER;
28      }
29  
30      private Win32BaseBoard() {
31      }
32  
33      /**
34       * Queries the Baseboard description.
35       *
36       * @return Baseboard manufacturer, model, and related fields.
37       */
38      public static WmiResult<BaseBoardProperty> queryBaseboardInfo() {
39          WmiQuery<BaseBoardProperty> baseboardQuery = new WmiQuery<>(WIN32_BASEBOARD, BaseBoardProperty.class);
40          return Objects.requireNonNull(WmiQueryHandler.createInstance()).queryWMI(baseboardQuery);
41      }
42  }
