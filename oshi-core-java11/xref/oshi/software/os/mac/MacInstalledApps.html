5   package oshi.software.os.mac;
6   
7   import oshi.software.os.ApplicationInfo;
8   import oshi.util.Constants;
9   import oshi.util.ExecutingCommand;
10  import oshi.util.ParseUtil;
11  
12  import java.util.ArrayList;
13  import java.util.List;
14  import java.util.Map;
15  import java.util.HashMap;
16  
17  public final class MacInstalledApps {
18  
19      private static final String COLON = ":";
20  
21      private MacInstalledApps() {
22      }
23  
24      public static List<ApplicationInfo> queryInstalledApps() {
25          List<String> output = ExecutingCommand.runNative("system_profiler SPApplicationsDataType");
26          return parseMacAppInfo(output);
27      }
28  
29      private static List<ApplicationInfo> parseMacAppInfo(List<String> lines) {
30          List<ApplicationInfo> appInfoList = new ArrayList<>();
31          String appName = null;
32          Map<String, String> appDetails = null;
33          boolean collectingAppDetails = false;
34  
35          for (String line : lines) {
36              line = line.trim();
37  
38              // Check for app name, ends with ":"
39              if (line.endsWith(COLON)) {
40                  // When app and appDetails are not empty then we reached the next app, add it to the list
41                  if (appName != null && !appDetails.isEmpty()) {
42                      appInfoList.add(createAppInfo(appName, appDetails));
43                  }
44  
45                  // store app name and proceed with collecting app details
46                  appName = line.substring(0, line.length() - 1);
47                  appDetails = new HashMap<>();
48                  collectingAppDetails = true;
49                  continue;
50              }
51  
52              // Process app details
53              if (collectingAppDetails && line.contains(COLON)) {
54                  int colonIndex = line.indexOf(COLON);
55                  String key = line.substring(0, colonIndex).trim();
56                  String value = line.substring(colonIndex + 1).trim();
57                  appDetails.put(key, value);
58              }
59          }
60  
61          return appInfoList;
62      }
63  
64      private static ApplicationInfo createAppInfo(String name, Map<String, String> details) {
65          String obtainedFrom = ParseUtil.getValueOrUnknown(details, "Obtained from");
66          String signedBy = ParseUtil.getValueOrUnknown(details, "Signed by");
67          String vendor = (obtainedFrom.equals("Identified Developer")) ? signedBy : obtainedFrom;
68  
69          String lastModified = details.getOrDefault("Last Modified", Constants.UNKNOWN);
70          long lastModifiedEpoch = ParseUtil.parseDateToEpoch(lastModified, "dd/MM/yy, HH:mm");
71  
72          // Additional info map
73          Map<String, String> additionalInfo = new HashMap<>();
74          additionalInfo.put("Kind", ParseUtil.getValueOrUnknown(details, "Kind"));
75          additionalInfo.put("Location", ParseUtil.getValueOrUnknown(details, "Location"));
76          additionalInfo.put("Get Info String", ParseUtil.getValueOrUnknown(details, "Get Info String"));
77  
78          return new ApplicationInfo(name, ParseUtil.getValueOrUnknown(details, "Version"), vendor, lastModifiedEpoch,
79                  additionalInfo);
80      }
81  }
