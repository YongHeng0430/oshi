5   package oshi.util;
6   
7   import static org.hamcrest.MatcherAssert.assertThat;
8   import static org.hamcrest.Matchers.contains;
9   import static org.hamcrest.Matchers.containsInAnyOrder;
10  import static org.hamcrest.Matchers.hasSize;
11  import static org.hamcrest.Matchers.is;
12  
13  import java.util.Map;
14  
15  import org.junit.jupiter.api.Test;
16  import org.junit.jupiter.api.condition.EnabledOnOs;
17  import org.junit.jupiter.api.condition.OS;
18  
19  @EnabledOnOs(OS.LINUX)
20  public class ProcUtilsTest {
21  
22      @Test
23      void testRawNetNetstat() {
24          String resource = ProcUtilsTest.class.getResource("sample-proc-net-netstat.txt").getFile();
25  
26          Map<String, Map<String, Long>> results = ProcUtil.parseNestedStatistics(resource);
27  
28          assertThat(results.keySet(), containsInAnyOrder("TcpExt", "IpExt", "MPTcpExt", "BadExt", "MoreBadExt"));
29          assertThat(results.get("TcpExt").get("SyncookiesSent"), is(6L));
30          assertThat(results.get("TcpExt").get("TCPAODroppedIcmps"), is(3L));
31          assertThat(results.get("IpExt").get("InNoRoutes"), is(55L));
32          assertThat(results.get("MPTcpExt").get("MPCurrEstab"), is(1L));
33          assertThat(results.get("BadExt").get("One"), is(1L));
34          assertThat(results.get("MoreBadExt").get("Six"), is(6L));
35      }
36  
37      @Test
38      void testRawNetNetstatWithLimitedKeys() {
39          String resource = ProcUtilsTest.class.getResource("sample-proc-net-netstat.txt").getFile();
40  
41          Map<String, Map<String, Long>> results = ProcUtil.parseNestedStatistics(resource, "IpExt");
42  
43          assertThat(results.keySet(), contains("IpExt"));
44          assertThat(results.get("IpExt").get("InNoRoutes"), is(55L));
45      }
46  
47      @Test
48      void testRawNetSnmp() {
49          String resource = ProcUtilsTest.class.getResource("sample-proc-net-snmp.txt").getFile();
50  
51          Map<String, Map<String, Long>> results = ProcUtil.parseNestedStatistics(resource);
52  
53          assertThat(results.keySet(), containsInAnyOrder("Ip", "Icmp", "IcmpMsg", "Tcp", "Udp", "UdpLite"));
54          assertThat(results.get("Tcp").get("ActiveOpens"), is(1892L));
55          assertThat(results.get("Ip").get("OutTransmits"), is(66296L));
56          assertThat(results.get("Icmp").get("InMsgs"), is(184L));
57      }
58  
59      @Test
60      void testRawNetSnmp6() {
61          String resource = ProcUtilsTest.class.getResource("sample-proc-net-snmp6.txt").getFile();
62  
63          Map<String, Long> results = ProcUtil.parseStatistics(resource);
64  
65          assertThat(results.keySet(), hasSize(91));
66          assertThat(results.get("Ip6InReceives"), is(8026L));
67          assertThat(results.get("Ip6OutMcastOctets"), is(45957L));
68          assertThat(results.get("UdpLite6MemErrors"), is(1L));
69          assertThat(results.get("IndentedEntry"), is(37L));
70      }

72  }
